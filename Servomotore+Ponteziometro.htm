#include 				//La libreria "Servo" viene inclusa

Servo SERVO1;					//Oggetto "Servo" creato. 
								//Esso servirà per controllare il Servomotore. 		

const int POTENZ1 = A5;			//Una variabile "costante" dichiarata.
								//Permetto di connettere il potenziometro al pin analogico A5.

								//nota*:sembrerebbe più fluido/veloce il "cont int" rispetto all' "int".

int val;						//Una variabile "intero" dichiarata.
								//Essa di tipo dato assumerà  
								//solo i valori "interi" 
								//il quale viene sfruttato
								//per leggere il valore digitale in analogico, assunto da A5 nel corso del tempo dall'accensione.
								
void setup()
{
SERVO1.attach(9);				//collegamento/attaccamneto del servomotore, abilitato nel pin specifico
}

void loop()
{
val = analogRead(POTENZ1);		//"int val" legge in modo analogico il valore del potenziometro.
  								//Il "pin A5" resta nei valori intorno "dai 0 e ai 1023".
val = map(val, 0, 1023, 0, 180);//Il valore letto viene riscalato al succesivo valore.
  
								//Convertirsi da un intervallo(x)a un altro intervallo (x alla n)
  								//mediante alla compatibilità col servo(0-180).
  
  								//attenzione*:Se non si rispetta ai parametri il servo non riuscirà ad eseguire ai comandi.

SERVO1.write(val);				//la posizione del servo viene assunta dai valori di "val".

delay(30);					//permetto al servo di raggiungere a una certa posizione dandogli del tempo di risposta
}
