#include 				//La libreria "Servo" viene inclusa

Servo SERVO1;					//Oggetto "Servo1" creato. 
								//Esso servirà per controllare il Servomotore. 		
Servo SERVO2;					//Oggetto "Servo2" creato. 
								//Esso servirà per controllare il Servomotore. 		
Servo SERVO3;					//Oggetto "Servo3" creato. 
								//Esso servirà per controllare il Servomotore. 		
Servo SERVO4;					//Oggetto "Servo4" creato. 
								//Esso servirà per controllare il Servomotore. 		

const int POTENZ1 = A1;			//Una variabile "costante" dichiarata.
								//Permetto di connettere il potenziometro al pin analogico A1.
const int POTENZ2 = A2;			//Una variabile "costante" dichiarata.
								//Permetto di connettere il potenziometro al pin analogico A2.
const int POTENZ3 = A3;			//Una variabile "costante" dichiarata.
								//Permetto di connettere il potenziometro al pin analogico A3.
const int POTENZ4 = A4;			//Una variabile "costante" dichiarata.
								//Permetto di connettere il potenziometro al pin analogico A4.

								//nota*:sembrerebbe più fluido/veloce il "cont int" rispetto all' "int".

int val1;						//variabili "interi" dichiarate.
								//Essi di tipo dato assumerà  
								//solo i valori "interi" 
								//il quale viene sfruttato
								//per leggere il valore digitale in analogico, assunto da A5 nel corso del tempo dall'accensione.
							

int val2;
int val3;
int val4;


void setup()
{
SERVO1.attach(10);				//collegamento/attaccamneto del servomotore, abilitato nel pin specifico(10)
SERVO2.attach(11);				//collegamento/attaccamneto del servomotore, abilitato nel pin specifico(11)
SERVO3.attach(12);				//collegamento/attaccamneto del servomotore, abilitato nel pin specifico(12)
SERVO4.attach(13);				//collegamento/attaccamneto del servomotore, abilitato nel pin specifico(13)

}

void loop()
{

  								//Il "pin A1" resta nei valori intorno "dai 0 e ai 1023".
val1 = map(val1, 0, 1023, 0, 180);//Il valore letto viene riscalato al succesivo valore.
  
								//Convertirsi da un intervallo(x)a un altro intervallo (x alla n)
  								//mediante alla compatibilità col servo(0-180).
  
  								//attenzione*:Se non si rispetta ai parametri il servo non riuscirà ad eseguire ai comandi.

SERVO1.write(val1);				//la posizione del servo viene assunta dai valori di "val1".

  								//mediante alla compatibilità col servo(0-180).
  
  								//attenzione*:Se non si rispetta ai parametri il servo non riuscirà ad eseguire ai comandi.

SERVO2.write(val2);				//la posizione del servo viene assunta dai valori di "val2".

  								//Il "pin A3" resta nei valori intorno "dai 0 e ai 1023".
val3 = map(val3, 0, 1023, 0, 180);//Il valore letto viene riscalato al succesivo valore.
  
								//Convertirsi da un intervallo(x)a un altro intervallo (x alla n)
  								//mediante alla compatibilità col servo(0-180).
  
  								//attenzione*:Se non si rispetta ai parametri il servo non riuscirà ad eseguire ai comandi.

  								//Il "pin A4" resta nei valori intorno "dai 0 e ai 1023".
val4 = map(val4, 0, 1023, 0, 180);//Il valore letto viene riscalato al succesivo valore.
  
								//Convertirsi da un intervallo(x)a un altro intervallo (x alla n)
  								//mediante alla compatibilità col servo(0-180).
  
  								//attenzione*:Se non si rispetta ai parametri il servo non riuscirà ad eseguire ai comandi.
