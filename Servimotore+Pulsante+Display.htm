// Progetto: Pull_Down con Servo_Motore e Display_LCD

#include         // Libreria Servo

#define PULS1     2
#define PULS2     9


Servo SERVOMOTORE_1 ;
Servo SERVOMOTORE_2 ;


// === VARIE DICHIARAZIONI ===
int pinControllo_1 = 7  ;		
int pinControllo_2 = 13 ;


// === LIBRERIE  LCD ===

// _Per TINKERCAD, usa questa:_

//#include 
//Adafruit_LiquidCrystal lcd_1(32);


// _Per ARDUINO IDE, usa questa:_
//#include 
//LiquidCrystal_I2C lcd_1(0x27, 16, 2);


void setup()
{
  
  SERVOMOTORE_1.attach(pinControllo_1);		//Servo motore1 al commando di Arduino
  SERVOMOTORE_2.attach(pinControllo_2);		//Servo motore2 al commando di Arduino

  pinMode(PULS1, INPUT);
  pinMode(PULS2, INPUT);
  
  SERVOMOTORE_1.write(90);           		// Arduino legge lo stato del Pulsante 1
 											// posizione 90 gradi a Destra
  SERVOMOTORE_2.write(90);					// Arduino legge lo stato del Pulsante 2
  											// posizione 90 gradi a Destra
  
   // INIZIALIZZAZIONE DEL DISPLAY
   //lcd_1.init();             // DA USARE SU ARDUINO IDE
   lcd_1.begin(16,2);
   lcd_1.setBacklight(1);
   lcd_1.setCursor (5,0);
   lcd_1.print("LEGGO");
   lcd_1.setCursor (0,1);
   lcd_1.print("I TUOI COMMANDI");
   delay(1000);
   
}

void loop()
{
  
  
int ValPULS1 = digitalRead(PULS1);     // Dichiariamo una Variabile di tipo INTERO 
                                       // Arduino legge lo stato del Pulsante 1
int ValPULS2 = digitalRead(PULS2);     // Dichiariamo una Variabile di tipo INTERO 
 
    

  
   if (ValPULS1 == HIGH)  // se il valore del Pulsante 1
    {                     // è Alto, Arduino esegue
                          // L’istruzione di seguito
     
   lcd_1.clear();           		// Clear LCD 
   SERVOMOTORE_1.write(0);     		//Dice al servomotore di andare 
                        			//alla posizione 0 gradi
                       				//tutto a Destra, di solito
   lcd_1.setCursor (0,0);
   lcd_1.print("COMMANDI:");
   lcd_1.setCursor (5,1);
   lcd_1.print("DESTRA");
   delay(1000);
   
   }
     else{
SERVOMOTORE_1.write(90);			//Dice al servomotore di andare 
                       				//alla posizione 90 gradi
                        			//tutto a Sinistra, per tornare
   									//alla posizione iniziale 
  }
  
  

   if (ValPULS2 == HIGH) 				// se il valore del Pulsante 2
    {                     				// è Alto, Arduino esegue
                          				// L’istruzione di seguito
     
   lcd_1.clear();            			// Clear LCD 
   SERVOMOTORE_2.write(180);        	//Dice al servomotore di andare 
                       					//alla posizione 180 gradi
                        				//tutto a Sinistra, di solito
   lcd_1.setCursor (0,0);
   lcd_1.print("COMMANDI:");
   lcd_1.setCursor (5,1);
   lcd_1.print("SINISTRA");
   delay(1000);
   }
  else{
SERVOMOTORE_2.write(90);				//Dice al servomotore di andare 
                       					//alla posizione 90 gradi
                        				//tutto a Sinistra, per tornare
   										//alla posizione iniziale 
  }
    
   
}
