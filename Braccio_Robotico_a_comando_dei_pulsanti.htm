#include         // Libreria Servo

#define PULS1     2
#define PULS2     3
#define PULS3     4
#define PULS4     5
#define PULS5     6
#define PULS6     7
#define PULS7     8
#define PULS8     9



Servo SERVOMOTORE_1 ;
Servo SERVOMOTORE_2 ;
Servo SERVOMOTORE_3 ;
Servo SERVOMOTORE_4 ;



// === VARIE DICHIARAZIONI ===
int Segnale_Servo_1 = 11  ;	
int Segnale_Servo_2 = 10  ;
int Segnale_Servo_3 = 12  ;
int Segnale_Servo_4 = 13  ;





void setup()
{
  
  SERVOMOTORE_1.attach(Segnale_Servo_1);		//Servo Servomotore1 al commando di Arduino
  SERVOMOTORE_2.attach(Segnale_Servo_2);		//Servo Servomotore2 al commando di Arduino
  SERVOMOTORE_3.attach(Segnale_Servo_3);		//Servo Servomotore3 al commando di Arduino
  SERVOMOTORE_4.attach(Segnale_Servo_4);		//Servo Servomotore4 al commando di Arduino

  pinMode(PULS1, INPUT);
  pinMode(PULS2, INPUT);
  pinMode(PULS3, INPUT);  
  pinMode(PULS4, INPUT);  
  pinMode(PULS5, INPUT);  
  pinMode(PULS6, INPUT);  
  pinMode(PULS7, INPUT);  
  pinMode(PULS8, INPUT);  

  SERVOMOTORE_1.write(90);           		// Posizione 90 gradi al centro
  delay(1000);
  SERVOMOTORE_2.write(90);           		// Posizione 90 gradi al centro
  delay(1000);
  SERVOMOTORE_3.write(90);           		// Posizione 90 gradi al centro
  delay(1000);
  SERVOMOTORE_4.write(90);           		// Posizione 90 gradi al centro
  delay(1000);
}

void loop()
{
  

int ValPULS1 = digitalRead(PULS1);     
int ValPULS2 = digitalRead(PULS2);     
int ValPULS3 = digitalRead(PULS3);     
int ValPULS4 = digitalRead(PULS4);     
int ValPULS5 = digitalRead(PULS5);     
int ValPULS6 = digitalRead(PULS6);     
int ValPULS7 = digitalRead(PULS7);     
int ValPULS8 = digitalRead(PULS8);     

  

  
  if (ValPULS1 == HIGH)   
    {                     
   SERVOMOTORE_1.write(0);     		
   delay(1000);
   }
  
  if (ValPULS2 == HIGH)   
    {                     
   SERVOMOTORE_1.write(180);     	
   delay(1000);
   }
  
  
  
  
  
  
  if (ValPULS3 == HIGH)  
    {                     
   SERVOMOTORE_2.write(0);     		
   delay(1000);
   }
  
  if (ValPULS4 == HIGH)   
    {                     
   SERVOMOTORE_2.write(180);     	
   delay(1000);
   }
  
  
  
  
  
  
  if (ValPULS5 == HIGH)  
    {                     
   SERVOMOTORE_3.write(0);     		
   delay(1000);
   }
  
  if (ValPULS6 == HIGH)   
    {                     
   SERVOMOTORE_3.write(180);     	
   delay(1000);
   }
  
  
  
  
  
  
  if (ValPULS7 == HIGH)  
    {                     
   SERVOMOTORE_4.write(0);     		
   delay(1000);
   }
  
  if (ValPULS8 == HIGH)   
    {                     
   SERVOMOTORE_4.write(180);     	
   delay(1000);
   }
  
    
     
   
}
