void setup()
{
  
  SERVOMOTORE_1.attach(Segnale_Servo_1);		//Servo Servomotore1 al commando di Arduino

  SERVOMOTORE_3.attach(Segnale_Servo_3);		//Servo Servomotore3 al commando di Arduino
  SERVOMOTORE_4.attach(Segnale_Servo_4);		//Servo Servomotore4 al commando di Arduino

  pinMode(PULS1, INPUT);
  pinMode(PULS2, INPUT);
  pinMode(PULS3, INPUT);  
  pinMode(PULS4, INPUT);  
  pinMode(PULS5, INPUT);  
  pinMode(PULS6, INPUT);  
  pinMode(PULS7, INPUT);  
  pinMode(PULS8, INPUT);
