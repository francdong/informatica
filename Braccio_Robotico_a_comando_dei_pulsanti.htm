#include <Servo.h>

#define PULS1     2
#define PULS2     3
#define PULS3     4
#define PULS4     5
#define PULS5     6
#define PULS6     7
#define PULS7     8
#define PULS8     9

Servo SERVOMOTORE_1;
Servo SERVOMOTORE_2;
Servo SERVOMOTORE_3;
Servo SERVOMOTORE_4;

int Segnale_Servo_1 = 11;
int Segnale_Servo_2 = 10;
int Segnale_Servo_3 = 12;
int Segnale_Servo_4 = 13;

void setup()
{
  SERVOMOTORE_1.attach(Segnale_Servo_1);
  SERVOMOTORE_2.attach(Segnale_Servo_2);
  SERVOMOTORE_3.attach(Segnale_Servo_3);
  SERVOMOTORE_4.attach(Segnale_Servo_4);

  for(int p=2; p<=9; p++) pinMode(p, INPUT);

  SERVOMOTORE_1.write(90);
  delay(1000);
  SERVOMOTORE_2.write(90);
  delay(1000);
  SERVOMOTORE_3.write(90);
  delay(1000);
  SERVOMOTORE_4.write(90);
  delay(1000);
}

void loop()
{
  int V1=digitalRead(PULS1), V2=digitalRead(PULS2);
  int V3=digitalRead(PULS3), V4=digitalRead(PULS4);
  int V5=digitalRead(PULS5), V6=digitalRead(PULS6);
  int V7=digitalRead(PULS7), V8=digitalRead(PULS8);

  if(V1==HIGH){ SERVOMOTORE_1.write(0); delay(1000); }
  if(V2==HIGH){ SERVOMOTORE_1.write(180); delay(1000); }
  if(V3==HIGH){ SERVOMOTORE_2.write(0); delay(1000); }
  if(V4==HIGH){ SERVOMOTORE_2.write(180); delay(1000); }
  if(V5==HIGH){ SERVOMOTORE_3.write(0); delay(1000); }
  if(V6==HIGH){ SERVOMOTORE_3.write(180); delay(1000); }
  if(V7==HIGH){ SERVOMOTORE_4.write(0); delay(1000); }
  if(V8==HIGH){ SERVOMOTORE_4.write(180); delay(1000); }
}
